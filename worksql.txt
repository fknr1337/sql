create table if not exists singers 
	(id serial primary key, 
	 name varchar(40) not null unique);

create table if not exists albums 
	(id serial primary key, 
	singer_id integer references Singers (id) unique, 
	name varchar(40) not null, 
	year integer not null);
	
create table if not exists tracks
	(id serial primary key,
	name varchar(40) not null,
	duration numeric(5,2) check(duration > 0),
	album_id integer references Albums (id));

create table if not exists genre
	(id serial primary key,
	singer_id integer references Singers(id) unique,
	name varchar(40) not null);

create table genreSinger
	(singer_id integer references Singers(id),
	genre_id integer references Genre(id),
	 constraint pk primary key (singer_id, genre_id));

create table if not exists collection
	(id serial primary key,
	 name varchar(40) not null,
	 year integer not null);

create table collectionTracks
	(id integer references tracks(id),
	track_id integer,
	album_id integer references Albums(id),
	constraint pks primary key (track_id, album_id));
